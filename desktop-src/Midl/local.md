``` syntax

    local

    void MyLocalProc(void);

    object, 
    uuid(12345678-1234-1234-123456789ABC),

    void MyLocalObjectProc(void);

    uuid(87654321-1234-1234-123456789ABC)

    [local] void MyLocalProc(void);

}
```
