#endif

   if ( CPUInfo[0] >= 1 )

#if defined(__clang__) || defined(__GNUC__)

#else
        __cpuid(CPUInfo, 1);
#endif

       if ( CPUInfo[2] & 0x80000 )
           g_bSSE41 = true;
   }
#endif
}
