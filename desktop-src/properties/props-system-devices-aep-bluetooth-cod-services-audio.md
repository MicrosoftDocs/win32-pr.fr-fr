```
propertyDescription
   name = System.Devices.Aep.Bluetooth.Cod.Services.Audio
   shellPKey = PKEY_Devices_Aep_Bluetooth_Cod_Services_Audio
   formatID = 5FBD34CD-561A-412E-BA98-478A6B0FEF1D

   SearchInfo
```
